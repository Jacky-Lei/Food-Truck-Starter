createFoodtruck: function (foodtruck, callback1, callback2) {
  $.ajax({
    url: "api/foodtrucks/",
    method: "POST",
    data: {foodtruck: foodtruck},
    success: function (foodtruck) {
      ApiActions.receiveSingleFoodtruck(foodtruck);
      callback1 && callback1(foodtruck.id)
      callback2 && callback2(foodtruck.id);
    }
  });
},
